    vocab_size = 50257,
    seq_len    = 128,   # Context length
    d_model    = 128,   # Vector size
    n_heads    = 4,     # Attention heads
    n_layers   = 4,     # Layers block count
    d_ff       = 512    # FeedForward (4 * d_model)
)
model = MiniGPT(cfg)
